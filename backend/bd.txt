Project finance_tracker {
  database_type: "PostgreSQL"
  Note: "Personal finance tracker database model"
}

/* =========================
   CATALOG TABLES
========================= */

Table currencies {
  id bigint [pk, increment]
  code varchar(3) [not null, unique, note: "Example: PEN, USD"]
  name varchar(50) [not null]
  symbol varchar(5) [not null]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table account_types {
  id bigint [pk, increment]
  code varchar(30) [not null, unique, note: "CASH, BANK_ACCOUNT, DIGITAL_WALLET, CREDIT_CARD"]
  name varchar(50) [not null, unique, note: "Cash, Bank Account, Digital Wallet, Credit Card"]
  description varchar(255)
  is_active boolean [not null, default: true]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table transaction_types {
  id bigint [pk, increment]
  code varchar(30) [not null, unique, note: "INCOME, EXPENSE, INTERNAL_TRANSFER"]
  name varchar(80) [not null]
  description varchar(255)

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table categories {
  id bigint [pk, increment]
  parent_category_id bigint
  transaction_type_id bigint [not null]
  name varchar(80) [not null]
  description varchar(255)
  is_active boolean [not null, default: true]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp

  indexes {
    (transaction_type_id, parent_category_id, name) [unique]
  }
}

Table contact_types {
  id bigint [pk, increment]
  code varchar(30) [not null, unique, note: "PERSON, COMPANY, STORE, EMPLOYER, BANK, OTHER"]
  name varchar(50) [not null, unique, note: "Person, Company, Store, Employer, Bank, Other"]
  description varchar(255)

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

/* =========================
   MAIN TABLES
========================= */

Table users {
  id bigint [pk, increment]
  full_name varchar(120) [not null]
  email varchar(120) [not null, unique]
  password_hash varchar(255) [not null]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
  deleted_at timestamp
}

Table accounts {
  id bigint [pk, increment]
  user_id bigint [not null]
  account_type_id bigint [not null]
  currency_id bigint [not null]

  name varchar(80) [not null, note: "Example: Yape, Interbank, Cash Wallet"]
  description varchar(255)

  initial_balance decimal(14,2) [not null, default: 0]
  current_balance decimal(14,2) [not null, default: 0]

  is_active boolean [not null, default: true]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
  deleted_at timestamp

  indexes {
    (user_id, name) [unique]
  }
}

Table contacts {
  id bigint [pk, increment]
  user_id bigint [not null]
  contact_type_id bigint [not null]

  name varchar(120) [not null, note: "Example: Mom, Company, Supermarket, Interbank"]
  phone varchar(30)
  email varchar(120)
  notes varchar(255)

  is_active boolean [not null, default: true]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
  deleted_at timestamp

  indexes {
    (user_id, name) [unique]
  }
}

Table transactions {
  id bigint [pk, increment]
  user_id bigint [not null]
  transaction_type_id bigint [not null]
  category_id bigint

  source_account_id bigint
  destination_account_id bigint

  source_contact_id bigint
  destination_contact_id bigint

  amount decimal(14,2) [not null]
  currency_id bigint [not null]

  transaction_date date [not null]
  description varchar(255)
  reference_code varchar(80)

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
  deleted_at timestamp

  indexes {
    (user_id, transaction_date)
    (source_account_id)
    (destination_account_id)
    (category_id)
  }
}

Table transaction_attachments {
  id bigint [pk, increment]
  transaction_id bigint [not null]

  file_name varchar(150) [not null]
  file_url varchar(500) [not null]
  file_type varchar(50)

  created_at timestamp [not null, default: `now()`]
}

/* =========================
   BUDGET / PLANNING TABLES
========================= */

Table budgets {
  id bigint [pk, increment]
  user_id bigint [not null]
  category_id bigint [not null]
  currency_id bigint [not null]

  name varchar(100) [not null]
  amount_limit decimal(14,2) [not null]

  start_date date [not null]
  end_date date [not null]

  is_active boolean [not null, default: true]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table saving_goals {
  id bigint [pk, increment]
  user_id bigint [not null]
  currency_id bigint [not null]

  name varchar(100) [not null]
  target_amount decimal(14,2) [not null]
  current_amount decimal(14,2) [not null, default: 0]

  target_date date
  is_completed boolean [not null, default: false]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

/* =========================
   RELATIONSHIPS
========================= */

Ref: accounts.user_id > users.id
Ref: accounts.account_type_id > account_types.id
Ref: accounts.currency_id > currencies.id

Ref: contacts.user_id > users.id
Ref: contacts.contact_type_id > contact_types.id

Ref: categories.parent_category_id > categories.id
Ref: categories.transaction_type_id > transaction_types.id

Ref: transactions.user_id > users.id
Ref: transactions.transaction_type_id > transaction_types.id
Ref: transactions.category_id > categories.id
Ref: transactions.currency_id > currencies.id

Ref: transactions.source_account_id > accounts.id
Ref: transactions.destination_account_id > accounts.id

Ref: transactions.source_contact_id > contacts.id
Ref: transactions.destination_contact_id > contacts.id

Ref: transaction_attachments.transaction_id > transactions.id

Ref: budgets.user_id > users.id
Ref: budgets.category_id > categories.id
Ref: budgets.currency_id > currencies.id

Ref: saving_goals.user_id > users.id
Ref: saving_goals.currency_id > currencies.id
